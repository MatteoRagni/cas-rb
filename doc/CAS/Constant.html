    # File lib/numbers/constants.rb, line 39
def simplify
  case @x
  when 0
    return CAS::Zero
  when 1
    return CAS::One
  when Math::PI
    return CAS::Pi
  when Math::E
    return CAS::E
  when 1.0/0.0
    return CAS::Infinity
  else
    return self
  end
end
