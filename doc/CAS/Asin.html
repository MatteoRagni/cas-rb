def diff(v)
  if @x.depend? v
    return @x.diff(v) / CAS.sqrt(CAS::One - CAS.pow(@x, CAS::Two))
  else
    return CAS::Zero
  end
end

def simplify
  super
  if @x == CAS::Zero
    return CAS::Zero
  end
  if @x == CAS::One
    return CAS::Pi / 2
  end
  if @x.is_a? CAS::Sin
    return @x.x
  end
  return self
end
